019/**
020 * Export metrics via the Prometheus Pushgateway.
021 * <p>
022 * The Prometheus Pushgateway exists to allow ephemeral and batch jobs to expose their metrics to Prometheus.
023 * Since these kinds of jobs may not exist long enough to be scraped, they can instead push their metrics
024 * to a Pushgateway. This class allows pushing the contents of a {@link CollectorRegistry} to
025 * a Pushgateway.
026 * <p>
027 * Example usage:
028 * <pre>
029 * {@code
030 *   void executeBatchJob() throws Exception {
031 *     CollectorRegistry registry = new CollectorRegistry();
032 *     Gauge duration = Gauge.build()
033 *         .name("my_batch_job_duration_seconds").help("Duration of my batch job in seconds.").register(registry);
034 *     Gauge.Timer durationTimer = duration.startTimer();
035 *     try {
036 *       // Your code here.
037 *
038 *       // This is only added to the registry after success,
039 *       // so that a previous success in the Pushgateway isn't overwritten on failure.
040 *       Gauge lastSuccess = Gauge.build()
041 *           .name("my_batch_job_last_success").help("Last time my batch job succeeded, in unixtime.").register(registry);
042 *       lastSuccess.setToCurrentTime();
043 *     } finally {
044 *       durationTimer.setDuration();
045 *       PushGateway pg = new PushGateway("127.0.0.1:9091");
046 *       pg.pushAdd(registry, "my_batch_job");
047 *     }
048 *   }
049 * }
050 * </pre>
051 * <p>
052 * See <a href="https://github.com/prometheus/pushgateway">https://github.com/prometheus/pushgateway</a>
053 */
054public class PushGateway {

245  void doRequest(CollectorRegistry registry, String job, Map<String, String> groupingKey, String method) throws IOException {
246    String url = "http://" + address + "/metrics/job/" + URLEncoder.encode(job, "UTF-8");
247    if (groupingKey != null) {
248      for (Map.Entry<String, String> entry: groupingKey.entrySet()) {
249        url += "/" + entry.getKey() + "/" + URLEncoder.encode(entry.getValue(), "UTF-8");
250      }
251    }
252    HttpURLConnection connection = (HttpURLConnection) new URL(url).openConnection();
253    connection.setRequestProperty("Content-Type", TextFormat.CONTENT_TYPE_004);
254    if (!method.equals("DELETE")) {
255      connection.setDoOutput(true);
256    }
257    connection.setRequestMethod(method);
258
259    connection.setConnectTimeout(10 * SECONDS_PER_MILLISECOND);
260    connection.setReadTimeout(10 * SECONDS_PER_MILLISECOND);
261    connection.connect();
262
263    try {
264      if (!method.equals("DELETE")) {
265        BufferedWriter writer = new BufferedWriter(new OutputStreamWriter(connection.getOutputStream()));
266        TextFormat.write004(writer, registry.metricFamilySamples());
267        writer.flush();
268        writer.close();
269      }
270
271      int response = connection.getResponseCode();
272      if (response != HttpURLConnection.HTTP_ACCEPTED) {
273        throw new IOException("Response code from " + url + " was " + response);
274      }
275    } finally {
276      connection.disconnect();
277    }
278  }

280  /**
281   * Returns a grouping key with the instance label set to the machine's IP address.
282   * <p>
283   * This is a convenience function, and should only be used where you want to
284   * push per-instance metrics rather than cluster/job level metrics.
285  */
286  public static Map<String, String> instanceIPGroupingKey() throws UnknownHostException {
287    Map<String, String> groupingKey = new HashMap<String, String>();
288    groupingKey.put("instance", InetAddress.getLocalHost().getHostAddress());
289    return groupingKey;
290  }
