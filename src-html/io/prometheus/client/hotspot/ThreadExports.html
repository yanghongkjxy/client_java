001package io.prometheus.client.hotspot;
002
003import io.prometheus.client.Collector;
004
005import java.lang.management.ManagementFactory;
006import java.lang.management.ThreadMXBean;
007import java.util.ArrayList;
008import java.util.Collections;
009import java.util.List;
010
011/**
012 * Exports metrics about JVM thread areas.
013 * <p>
014 * Example usage:
015 * <pre>
016 * {@code
017 *   new ThreadExports().register();
018 * }
019 * </pre>
020 * Example metrics being exported:
021 * <pre>
022 *   jvm_threads_current{} 300
023 *   jvm_threads_daemon{} 200
024 *   jvm_threads_peak{} 410
025 *   jvm_threads_started_total{} 1200
026 * </pre>
027 */
028public class ThreadExports extends Collector {
029  private final ThreadMXBean threadBean;
030
031  public static final List<String> EMPTY_LABEL = Collections.emptyList();
032
033  public ThreadExports() {
034    this(ManagementFactory.getThreadMXBean());
035  }
036
037  public ThreadExports(ThreadMXBean threadBean) {
038    this.threadBean = threadBean;
039  }
040
041  void addThreadMetrics(List<MetricFamilySamples> sampleFamilies) {
042    sampleFamilies.add(
043            new MetricFamilySamples(
044                    "jvm_threads_current",
045                    Type.GAUGE,
046                    "Current thread count of a JVM",
047                    Collections.singletonList(
048                            new MetricFamilySamples.Sample(
049                                    "jvm_threads_current", EMPTY_LABEL, EMPTY_LABEL,
050                                    threadBean.getThreadCount()))));
051
052    sampleFamilies.add(
053            new MetricFamilySamples(
054                    "jvm_threads_daemon",
055                    Type.GAUGE,
056                    "Daemon thread count of a JVM",
057                    Collections.singletonList(
058                            new MetricFamilySamples.Sample(
059                                    "jvm_threads_daemon", EMPTY_LABEL, EMPTY_LABEL,
060                                    threadBean.getDaemonThreadCount()))));
061
062    sampleFamilies.add(
063            new MetricFamilySamples(
064                    "jvm_peak_threads",
065                    Type.GAUGE,
066                    "Peak thread count of a JVM",
067                    Collections.singletonList(
068                            new MetricFamilySamples.Sample(
069                                    "jvm_threads_peak", EMPTY_LABEL, EMPTY_LABEL,
070                                    threadBean.getPeakThreadCount()))));
071
072    sampleFamilies.add(
073            new MetricFamilySamples(
074                    "jvm_threads_started_total",
075                    Type.COUNTER,
076                    "Started thread count of a JVM",
077                    Collections.singletonList(
078                            new MetricFamilySamples.Sample(
079                                    "jvm_threads_started_total", EMPTY_LABEL, EMPTY_LABEL,
080                                    threadBean.getTotalStartedThreadCount()))));
081  }
082
083
084  public List<MetricFamilySamples> collect() {
085    List<MetricFamilySamples> mfs = new ArrayList<MetricFamilySamples>();
086    addThreadMetrics(mfs);
087    return mfs;
088  }
089}
