001package io.prometheus.client.logback;
002
003import ch.qos.logback.classic.Level;
004import ch.qos.logback.classic.spi.ILoggingEvent;
005import ch.qos.logback.core.UnsynchronizedAppenderBase;
006import io.prometheus.client.Counter;
007
008public class InstrumentedAppender extends UnsynchronizedAppenderBase<ILoggingEvent> {
009
010  public static final String COUNTER_NAME = "logback_appender_total";
011  
012  private static final Counter COUNTER;
013  private static final Counter.Child TRACE_LABEL;
014  private static final Counter.Child DEBUG_LABEL;
015  private static final Counter.Child INFO_LABEL;
016  private static final Counter.Child WARN_LABEL;
017  private static final Counter.Child ERROR_LABEL;
018  
019  static {
020    COUNTER = Counter.build().name(COUNTER_NAME)
021            .help("Log4j log statements at various log levels")
022            .labelNames("level")
023            .register();
024
025    TRACE_LABEL = COUNTER.labels("trace");
026    DEBUG_LABEL = COUNTER.labels("debug");
027    INFO_LABEL = COUNTER.labels("info");
028    WARN_LABEL = COUNTER.labels("warn");
029    ERROR_LABEL = COUNTER.labels("error");
030  }
031
032  /**
033   * Create a new instrumented appender using the default registry.
034   */
035  public InstrumentedAppender() {
036  }
037
038
039  @Override
040  public void start() {
041    super.start();
042  }
043
044  @Override
045  protected void append(ILoggingEvent event) {
046    switch (event.getLevel().toInt()) {
047      case Level.TRACE_INT:
048        TRACE_LABEL.inc();
049        break;
050      case Level.DEBUG_INT:
051        DEBUG_LABEL.inc();
052        break;
053      case Level.INFO_INT:
054        INFO_LABEL.inc();
055        break;
056      case Level.WARN_INT:
057        WARN_LABEL.inc();
058        break;
059      case Level.ERROR_INT:
060        ERROR_LABEL.inc();
061        break;
062      default:
063        break;
064    }
065  }
066}
